class Solution {
    public int lengthOfLastWord(String s) {
        
        int i = s.length() - 1;
        int count = 0;

        // Step 1: Skip ending spaces
        while (i >= 0 && s.charAt(i) == ' ') {
            i--;
        }

        // Step 2: Count characters of last word
        while (i >= 0 && s.charAt(i) != ' ') {
            count++;
            i--;
        }

        return count;
    }
}
