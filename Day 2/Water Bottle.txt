6.Water Bottle
------------------------------

solution:

class Solution {
public:
    int numWaterBottles(int numBottles, int numExchange) {
            int ans= numBottles;

            while(numBottles >= numExchange)
            {
                int newBottle = numBottles/numExchange;
                int remBottles = numBottles%numExchange;
                ans= ans + newBottle;
                numBottles = remBottles + newBottle;
            }
            return ans;   
    }
};
